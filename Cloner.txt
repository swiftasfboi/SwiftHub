---------------------
-- DRAGGING (top bar)
---------------------
local dragging = false
local dragStart
local startPos

top.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragging = true
		dragStart = input.Position
		startPos = main.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

UIS.InputChanged:Connect(function(input)
	if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
		local delta = input.Position - dragStart
		main.Position = UDim2.new(
			startPos.X.Scale,
			startPos.X.Offset + delta.X,
			startPos.Y.Scale,
			startPos.Y.Offset + delta.Y
		)
	end
end)

---------------------
-- HIDE / OPEN button logic (smooth)
---------------------
local function hideUI()
	TweenService:Create(main, TweenInfo.new(0.22, Enum.EasingStyle.Quad), {
		Position = UDim2.new(main.Position.X.Scale, main.Position.X.Offset, 1.15, 0),
		BackgroundTransparency = 1
	}):Play()
	task.wait(0.22)
	main.Visible = false
	openBtn.Visible = true
end

local function showUI()
	main.Visible = true
	openBtn.Visible = false
	TweenService:Create(main, TweenInfo.new(0.22, Enum.EasingStyle.Quad), {
		Position = UDim2.new(0.5, -190, 0.5, -160),
	}):Play()
end

hide.MouseButton1Click:Connect(hideUI)
openBtn.MouseButton1Click:Connect(showUI)
